 15def plot_sumfile(
 16    ax,
 17    time,
 18    dp,
 19    dndlogdp,
 20    clim=(10,100000),
 21    cmap='jet',
 22    cscale='log',
 23    cbar=True,
 24    cbar_padding=0.05):    
 25    """ 
 26    Plot aerosol number-size distribution surface plot
 27
 28    Parameters
 29    ----------
 30
 31    ax : axes object
 32        axis on which to plot the data
 33
 34    time : numpy 1d array, size n
 35        measurement times
 36
 37    dp : numpy 1d array, size m 
 38        particle diameters
 39
 40    dndlogdp : numpy 2d array, size (n,m)
 41        number-size distribution matrix
 42
 43    clim : iterable with two numerical elements
 44        color limits
 45
 46    cmap : str 
 47        colormap to be used
 48
 49    cscale : str         
 50        type of color scale `log` or `linear`
 51        
 52    cbar : boolean
 53        plot colorbar `True` or `False`
 54    
 55    cbar_padding : float
 56        Control space between colorbar and axes
 57    
 58    Returns
 59    -------
 60
 61    matplotlib.collections.QuadMesh
 62        
 63    Colorbar
 64        only if cbar is `True`
 65
 66    """
 67    
 68    mesh_dp,mesh_time = np.meshgrid(dp,time)
 69
 70    if cscale=='log':
 71        norm = colors.LogNorm()
 72        ticks = LogLocator(subs=range(10))
 73    elif cscale=='linear':
 74        norm = None
 75        ticks = None
 76    else:
 77        raise Exception('"cscale" must be "log" or "linear"') 
 78
 79    pcolorplot = ax.pcolormesh(

 81        mesh_dp,
 82        dndlogdp,
 83        norm=norm,
 84        linewidth=0,
 85        rasterized=True,
 86        cmap=cmap)
 87
 88    pcolorplot.set_clim(clim)
 89    ax.set_yscale('log')
 90    ax.autoscale(tight='true')
 91    ax.grid(which="both",c='k',ls=':',alpha=0.5)
 92
 93    if cbar:
 94        cbar_ax = plt.colorbar(
 95            pcolorplot,
 96            ax=ax,
 97            pad=cbar_padding,
 98            ticks=ticks)
 99        return pcolorplot, cbar_ax

104def dndlogdp2dn(dp,dndlogdp):
105    """    
106    Convert from normalized number concentrations to
107    unnormalized number concentrations assuming that 
108    the size channels have common edges.
109
110    Parameters
111    ----------
112
113    dp : numpy 1d array
114        Geometric mean diameters for the size channels
115
116    dndlogdp : numpy 2d array
117        Number size distribution with normalized concentrations
118        i.e. dN/dlogDp
119
120    Returns
121    -------
122
123    2-d array
124        The number size distribution with unnormalized concentrations 
125        i.e. dN
126
127    """
128
129    logdp_mid = np.log10(dp)
130    logdp = (logdp_mid[:-1]+logdp_mid[1:])/2.0
131    logdp = np.append(logdp,logdp_mid.max()+(logdp_mid.max()-logdp.max()))
132    logdp = np.insert(logdp,0,logdp_mid.min()-(logdp.min()-logdp_mid.min()))
133    dlogdp = np.diff(logdp)
134    return dndlogdp*dlogdp

137def air_viscosity(temp):
138    """ 
139    Calculate air viscosity
140    using Enskog-Chapman theory
141
142    Parameters
143    ----------
144
145    temp : float or array
146        air temperature, unit: K  
147
148    Returns
149    -------
150
151    float or array
152        viscosity of air, unit: m2 s-1  
153
154    """
155
156    nyy_ref=18.203e-6
157    S=110.4
158    temp_ref=293.15
159    return nyy_ref*((temp_ref+S)/(temp+S))*((temp/temp_ref)**(3./2.))

161def mean_free_path(temp,pres):
162    """ 
163    Calculate mean free path in air
164
165    Parameters
166    ----------
167
168    temp : float
169        air temperature, unit: K  
170
171    pres : float
172        air pressure, unit: Pa
173
174    Returns
175    -------
176
177    float
178        mean free path in air, unit: m  
179
180    """
181
182    R=8.3143
183    Mair=0.02897
184    mu=air_viscosity(temp)
185    return (2.*mu)/(pres*(8.*Mair/(np.pi*R*temp))**(1./2.))

187def slipcorr(dp,temp,pres):
188    """
189    Slip correction factor in air 
190
191    Parameters
192    ----------
193
194    dp : float or numpy array
195        particle diameter, unit: m 
196
197    temp : float
198        air temperature, unit: K 
199
200    pres : float
201        air pressure, unit: Pa
202
203    Returns
204    -------
205
206    float or numpy array
207        Cunningham slip correction factor for each particle diameter, 
208        unit: dimensionless        
209
210    """
211   
212    l = mean_free_path(temp,pres)
213    return 1.+((2.*l)/dp)*(1.257+0.4*np.exp(-(1.1*dp)/(2.*l)))

215def particle_diffusivity(dp,temp,pres):
216    """ 
217    Particle brownian diffusivity in air 
218
219    Parameters
220    ----------
221
222    dp : float or array
223        particle diameter, unit: m 
224
225    temp : float
226        air temperature, unit: K 
227
228    pres : float
229        air pressure, unit: Pa
230
231    Returns
232    -------
233
234    float or array
235        Brownian diffusivity in air for particles of size dp,
236        unit: m2 s-1
237
238    """
239
240    k=1.381e-23
241    cc=slipcorr(dp,temp,pres)
242    mu=air_viscosity(temp)
243
244    return (k*temp*cc)/(3.*np.pi*mu*dp)

246def particle_thermal_speed(dp,temp):
247    """
248    Particle thermal speed 
249
250    Parameters
251    ----------
252
253    dp : float or array
254        particle diameter, unit: m 
255
256    temp : float
257        air temperature, unit: K 
258
259    Returns
260    -------
261
262    float or array
263        Particle thermal speed for each dp, unit: m s-1
264
265    """
266
267    k=1.381e-23
268    rho_p=1000.0
269    mp=rho_p*(1./6.)*np.pi*dp**3.
270    
271    return ((8.*k*temp)/(np.pi*mp))**(1./2.)

273def particle_mean_free_path(dp,temp,pres):
274    """ 
275    Particle mean free path in air 
276
277    Parameters
278    ----------
279
280    dp : float or array
281        particle diameter, unit: m 
282
283    temp : float
284        air temperature, unit: K 
285
286    pres : float
287        air pressure, unit: Pa
288
289    Returns
290    -------
291
292    float or array
293        Particle mean free path for each dp, unit: m
294
295    """
296
297    D=particle_diffusivity(dp,temp,pres)
298    c=particle_thermal_speed(dp,temp)
299
300    return (8.*D)/(np.pi*c)

302def coagulation_coef(dp1,dp2,temp,pres):
303    """ 
304    Calculate Brownian coagulation coefficient (Fuchs)
305
306    Parameters
307    ----------
308
309    dp1 : float
310        first particle diameter, unit: m 
311
312    dp2 : float
313        second particle diameter, unit: m 
314
315    temp : float
316        air temperature, unit: K 
317
318    pres : float
319        air pressure, unit: Pa
320
321    Returns
322    -------
323
324    float or array
325        Brownian coagulation coefficient (Fuchs), unit: m3 s-1
326
327    """
328
329    def particle_g(dp,temp,pres):
330        l = particle_mean_free_path(dp,temp,pres)    
331        return 1./(3.*dp*l)*((dp+l)**3.-(dp**2.+l**2.)**(3./2.))-dp
332
333    D1 = particle_diffusivity(dp1,temp,pres)
334    D2 = particle_diffusivity(dp2,temp,pres)
335    g1 = particle_g(dp1,temp,pres)
336    g2 = particle_g(dp2,temp,pres)
337    c1 = particle_thermal_speed(dp1,temp)
338    c2 = particle_thermal_speed(dp2,temp)
339    
340    return 2.*np.pi*(D1+D2)*(dp1+dp2) \
341           * ( (dp1+dp2)/(dp1+dp2+2.*(g1**2.+g2**2.)**0.5) + \
342           +   (8.*(D1+D2))/((c1**2.+c2**2.)**0.5*(dp1+dp2)) )

344def calc_coags(Dp,dp,dndlogdp,temp,pres):
345    """ 
346    Calculate coagulation sink
347
348    Kulmala et al (2012): doi:10.1038/nprot.2012.091 
349
350    Parameters
351    ----------
352
353    Dp : float
354        Particle diameter for which you want to calculate the CoagS, 
355        unit: m
356
357    dp : numpy 1d array, size m
358        diameter in the data, unit: meters,
359        unit: m
360
361    dndlogdp : numpy 2d array, size (n,m)
362        dN/dlogDp matrix,
363        unit: cm-3
364
365    temp : float or numpy 1d array of size n
366        Ambient temperature corresponding to the data,
367        unit: K
368
369    pres : float or numpy 1d array of size n
370        Ambient pressure corresponding to the data,
371        unit: Pa
372
373    Returns
374    -------
375    
376    numpy 1d array, size n
377        Coagulation sink time series,
378        unit: s-1
379
380    """
381
382    n = dndlogdp.shape[0]
383
384    if not isinstance(temp,Iterable):
385        temp = temp*np.ones(n)
386
387    if not isinstance(pres,Iterable):
388        pres = pres*np.ones(n)
389
390    dn = dndlogdp2dn(dp,dndlogdp)
391    dp = dp[dp>=Dp]
392    dn = dn[:,dp>=Dp]
393
394    coags = np.nan*np.ones(n)
395
396    for i in range(n):
397        # multiply by 1e6 to make [K] = cm3 s-1
398        coags[i] = np.nansum(1e6*coagulation_coef(Dp,dp,temp[i],pres[i])*dn[i,:])
399                
400    return coags

402def diam2mob(dp,temp,pres,ne):
403    """ 
404    Convert electrical mobility diameter to electrical mobility in air
405
406    Parameters
407    ----------
408
409    dp : float or numpy 1d array
410        particle diameter(s),
411        unit : m
412
413    temp : float
414        ambient temperature, 
415        unit: K
416
417    pres : float
418        ambient pressure, 
419        unit: Pa
420
421    ne : int
422        number of charges on the aerosol particle
423
424    Returns
425    -------
426
427    float or numpy 1d array
428        particle electrical mobility or mobilities, 
429        unit: m2 s-1 V-1
430
431    """
432
433    e = 1.60217662e-19
434    cc = slipcorr(dp,temp,pres)
435    mu = air_viscosity(temp)
436
437    Zp = (ne*e*cc)/(3.*np.pi*mu*dp)
438
439    return Zp

441def mob2diam(Zp,temp,pres,ne):
442    """
443    Convert electrical mobility to electrical mobility diameter in air
444
445    Parameters
446    ----------
447
448    Zp : float or numpy 1d array
449        particle electrical mobility or mobilities, 
450        unit: m2 s-1 V-1
451
452    temp : float
453        ambient temperature, 
454        unit: K
455
456    pres : float
457        ambient pressure, 
458        unit: Pa
459
460    ne : integer
461        number of charges on the aerosol particle
462
463    Returns 
464    -------
465
466    float or numpy 1d array
467        particle diameter(s), 
468        unit: m
469
470    """
471
472    def minimize_this(dp,Z):
473        return np.abs(diam2mob(dp,temp,pres,ne)-Z)
474
475    dp0 = 0.0001
476
477    result = minimize(minimize_this, dp0, args=(Zp,), tol=1e-20, method='Nelder-Mead').x[0]    
478
479    return result

481def binary_diffusivity(temp,pres,Ma,Mb,Va,Vb):
482    """ 
483    Binary diffusivity in a mixture of gases a and b
484
485    Fuller et al. (1966): https://doi.org/10.1021/ie50677a007 
486
487    Parameters
488    ----------
489
490    temp : float
491        temperature, 
492        unit: K
493
494    pres : float
495        pressure, 
496        unit: Pa
497
498    Ma : float
499        relative molecular mass of gas a, 
500        unit: dimensionless
501
502    Mb : float
503        relative molecular mass of gas b, 
504        unit: dimensionless
505
506    Va : float
507        diffusion volume of gas a, 
508        unit: dimensionless
509
510    Vb : float
511        diffusion volume of gas b, 
512        unit: dimensionless
513
514    Returns
515    -------
516
517    float
518        binary diffusivity, 
519        unit: m2 s-1
520
521    """
522    
523    # convert pressure to atmospheres
524
525    diffusivity = (1.013e-2*(temp**1.75)*np.sqrt((1./Ma)+(1./Mb)))/(pres*(Va**(1./3.)+Vb**(1./3.))**2)
526    return diffusivity

529def beta(dp,temp,pres,diffusivity,molar_mass):
530    """ 
531    Calculate Fuchs Sutugin correction factor 
532
533    Sutugin et al. (1971): https://doi.org/10.1016/0021-8502(71)90061-9
534
535    Parameters
536    ----------
537
538    dp : float or numpy 1d array
539        aerosol particle diameter(s), 
540        unit: m
541
542    temp : float
543        temperature, 
544        unit: K
545
546    pres : float
547        pressure,
548        unit: Pa
549
550    diffusivity : float
551        diffusivity of the gas that is condensing, 
552        unit: m2/s
553
554    molar_mass : float
555        molar mass of the condensing gas, 
556        unit: g/mol
557
558    Returns
559    -------
560
561    float or 1-d numpy array
562        Fuchs Sutugin correction factor for each particle diameter, 
563        unit: m2/s
564
565    """
566
567    R = 8.314 
568    l = 3.*diffusivity/((8.*R*temp)/(np.pi*molar_mass*0.001))**0.5
569    knud = 2.*l/dp
570    
571    return (1. + knud)/(1. + 1.677*knud + 1.333*knud**2)

573def calc_cs(dp,dndlogdp,temp,pres):
574    """
575    Calculate condensation sink, assuming that the condensing gas is sulfuric acid in air
576    with aerosol particles.
577
578    Kulmala et al (2012): doi:10.1038/nprot.2012.091 
579
580    Parameters
581    ----------
582
583    dp : numpy 1d array, size m
584        diameter in the data, unit: m
585
586    dndlogdp : numpy 2d array, size (n,m)
587        dN/dlogDp matrix, unit: cm-3
588
589    temp : numpy 1d array, size n
590        Ambient temperature corresponding to the data, unit: K
591
592    pres : numpy 1d array, size n
593        Ambient pressure corresponding to the data, unit: Pa
594
595    Returns
596    -------
597    
598    numpy 1d array, size n
599        condensation sink time series, unit: s-1
600
601    """
602
603    n = dndlogdp.shape[0]
604
605    if not isinstance(temp,Iterable):
606        temp=temp*np.ones(n)
607
608    if not isinstance(pres,Iterable):
609        pres=pres*np.ones(n)
610
611    M_h2so4 = 98.08   
612    M_air = 28.965    
613    V_air = 19.7      
614    V_h2so4 = 51.96  
615
616    dn = dndlogdp2dn(dp,dndlogdp)
617    cs = np.nan*np.ones(n)
618
619    for i in range(n):
620        diffusivity = binary_diffusivity(temp[i],pres[i],M_h2so4,M_air,V_h2so4,V_air)
621        b = beta(dp,temp[i],pres[i],diffusivity,M_h2so4)
622
623        cs[i] = (4.*np.pi*diffusivity)*np.nansum(1e6*dn[i,:]*b*dp)
624
625    return cs

627def calc_conc(dp,dndlogdp,dmin,dmax):
628    """
629    Calculate particle number concentration from aerosol 
630    number-size distribution
631
632    Parameters
633    ----------
634
635    dp : numpy 1d array, size m
636        diameter in the data, unit: m
637
638    dndlogdp : numpy 2d array, size (n,m)
639        dN/dlogDp matrix, unit: cm-3
640
641    dmin : float
642        Size range lower diameter, unit: m
643
644    dmax : float
645        Size range upper diameter, unit: m
646
647    Returns
648    -------
649    
650    numpy 1d array, size n
651        Number concentration in the given size range, unit: cm-3
652
653    """
654    
655    findex = np.argwhere((dp<=dmax)&(dp>=dmin)).flatten()
656    dp = dp[findex]
657    dndlogdp = dndlogdp[:,findex]
658    logdp_mid = np.log10(dp)
659    logdp = (logdp_mid[:-1]+logdp_mid[1:])/2.0
660    logdp = np.append(logdp,logdp_mid.max()+(logdp_mid.max()-logdp.max()))
661    logdp = np.insert(logdp,0,logdp_mid.min()-(logdp.min()-logdp_mid.min()))
662    dlogdp = np.diff(logdp)
663    return np.nansum(dndlogdp*dlogdp,axis=1)

665def calc_formation_rate(time,dp1,dp2,conc,coags,gr):
666    """
667    Calculate particle formation rate
668
669    Kulmala et al (2012): doi:10.1038/nprot.2012.091
670
671    Parameters
672    ----------
673
674    time : numpy 1d array
675        time associated with the measurements
676
677    dp1 : float
678        Lower diameter of the size range, unit: m
679
680    dp2 : float
681        Upper diameter of the size range, unit: m
682
683    conc : numpy 1d array
684        Particle number concentration in the size range dp1...dp2, unit: cm-3
685
686    coags : numpy 1d array
687        Coagulation sink for particles in the size range dp1...dp2. Usually approximated as coagulation sink for particle size dp1, unit: s-1
688
689    gr : float
690        Growth rate for particles out of the size range dp1...dp2, unit: nm h-1
691
692    Returns
693    -------
694
695    numpy 1d array
696        particle formation rate for diameter dp1, unit: cm3 s-1
697
698    """
699
700    conc_term = np.diff(conc)/np.diff(time*1.157e5)
701    sink_term = (coags[1:] + coags[:-1])/2. * (conc[1:] + conc[:-1])/2.
702    gr_term = (2.778e-13*gr)/(dp2-dp1) * (conc[1:] + conc[:-1])/2.
703    formation_rate = conc_term + sink_term + gr_term
704
705    return formation_rate

707def calc_ion_formation_rate(
708    time,
709    dp1,
710    dp2,
711    conc_pos,
712    conc_neg,
713    conc_pos_small,
714    conc_neg_small,
715    conc,
716    coags,
717    gr):
718    """ 
719    Calculate ion formation rate
720
721    Kulmala et al (2012): doi:10.1038/nprot.2012.091
722
723    Parameters
724    ----------
725
726    time : numpy 1d array
727        Time associated with the measurements, unit: days  
728
729    dp1 : float
730        Lower diameter of the size range, unit: m
731
732    dp2 : float
733        Upper diameter of the size range, unit: m
734
735    conc_pos : numpy 1d array
736        Positive ion number concentration in the size range dp1...dp2, unit: cm-3
737
738    conc_neg : numpy 1d array
739        Negative ion number concentration in the size range dp1...dp2, unit: cm-3
740
741    conc_pos_small : numpy 1d array
742        Positive ion number concentration for ions smaller than dp1, unit: cm-3
743
744    conc_neg_small : numpy 1d array
745        Negative ion number concentration for ions smaller than dp1, unit: cm-3
746
747    conc : numpy 1d array
748        Particle number concentration in the size range dp1...dp2, unit: cm-3
749
750    coags : numpy 1d array
751        Coagulation sink for particles in the size range dp1...dp2.
752        Usually approximated as coagulation sink for particle size dp1, 
753        unit: s-1
754
755    gr : float
756        Growth rate for particles out of the size range dp1...dp2, unit: nm h-1
757
758    Returns
759    -------
760
761    numpy 1d array
762        Positive ion formation rate for diameter dp1, unit : cm3 s-1
763
764    numpy 1d array
765        Negative ion formation rate for diameter dp1, unit: cm3 s-1
766
767    """
768
769    alpha = 1.6e-6 # cm3 s-1
770    Xi = 0.01e-6 # cm3 s-1
771
772    coags = (coags[1:] + coags[:-1])/2.
773    conc_pos = (conc_pos[1:] + conc_pos[:-1])/2.
774    conc_neg = (conc_neg[1:] + conc_neg[:-1])/2.
775    conc_pos_small = (conc_pos_small[1:] + conc_pos_small[:-1])/2.
776    conc_neg_small = (conc_neg_small[1:] + conc_neg_small[:-1])/2.
777    conc = (conc[1:] + conc[:-1])/2.
778
779    pos_conc_term = np.diff(conc_pos)/np.diff(time*1.157e5)
780    pos_sink_term = coags * conc_pos
781    pos_gr_term = (2.778e-13*gr)/(dp2-dp1) * conc_pos
782    pos_recombination_term = alpha * conc_pos * conc_neg_small
783    pos_charging_term = Xi * conc * conc_pos_small
784    pos_formation_rate = pos_conc_term + pos_sink_term + pos_gr_term + pos_recombination_term - pos_charging_term
785
786    neg_conc_term = np.diff(conc_neg)/np.diff(time*1.157e5)
787    neg_sink_term = coags * conc_neg
788    neg_gr_term = (2.778e-13*gr)/(dp2-dp1) * conc_neg
789    neg_recombination_term = alpha * conc_neg * conc_pos_small
790    neg_charging_term = Xi * conc * conc_neg_small
791    neg_formation_rate = neg_conc_term + neg_sink_term + neg_gr_term + neg_recombination_term - neg_charging_term
792
793    return pos_formation_rate, neg_formation_rate

795def datenum2datetime(datenum):
